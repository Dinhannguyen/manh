-- [[ SCRIPT 2: WATCHER & HOPPER - FINAL LOGIC ]] --
local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")

local folderName = "mm"
local LocalPlayer = Players.LocalPlayer

-- lastSafeTime = 0: Nghĩa là đang ở Server Rác (mặc định)
-- lastSafeTime > 0: Nghĩa là vừa mới ở Server Bảo Kê
local lastSafeTime = 0 

-- Hàm check Boss/Item (Giữ nguyên)
local function checkDoughKing()
    local enemies = Workspace:FindFirstChild("Enemies")
    if not enemies then return false end
    for _, model in pairs(enemies:GetChildren()) do
        if model.Name == "Dough King" and model:FindFirstChild("Humanoid") and model.Humanoid.Health > 0 then
            return true
        end
    end
    return false
end

local function amIHoldingChalice()
    local backpack = LocalPlayer:FindFirstChild("Backpack")
    local character = LocalPlayer.Character
    local items = {"God's Chalice", "Sweet Chalice"}
    for _, itemName in ipairs(items) do
        if backpack and backpack:FindFirstChild(itemName) then return true end
        if character and character:FindFirstChild(itemName) then return true end
    end
    return false
end

print("✅ Watcher đang chạy. Trạng thái: Server Rác (Sẵn sàng Hop)")

while true do
    wait(2) -- Quét mỗi 2 giây

    -- 1. Check an toàn (Nếu đang đánh Boss thì coi như đang ở Server Bảo Kê để không hop)
    if checkDoughKing() or amIHoldingChalice() then
        print("⛔ Có Boss/Chalice! Tạm dừng hop.")
        lastSafeTime = os.time() -- Set trạng thái an toàn giả
        continue
    end

    if not isfolder(folderName) then makefolder(folderName) end
    local files = listfiles(folderName)
    
    -- 2. Kiểm tra: Mình có đang ở Server Bảo Kê không?
    local isSafeServer = false
    for _, file in ipairs(files) do
        local success, content = pcall(readfile, file)
        if success then
            local data = HttpService:JSONDecode(content)
            if data.jobId == game.JobId then
                isSafeServer = true
                break
            end
        end
    end

    -- 3. XỬ LÝ LOGIC TRẠNG THÁI
    if isSafeServer then
        -- [TRƯỜNG HỢP A]: Đang ở đúng Server của file
        lastSafeTime = os.time() -- Cập nhật liên tục
        print("✅ [Safe] Đang ở Server Bảo Kê. (Không Hop)")
        continue -- Quay lại đầu, KHÔNG chạy xuống phần Hop
    
    elseif lastSafeTime ~= 0 then
        -- [TRƯỜNG HỢP B]: Vừa mất file (Trước đó là Safe, giờ mới mất)
        -- Logic này CHỈ chạy khi lastSafeTime khác 0
        local timePassed = os.time() - lastSafeTime
        
        if timePassed < 30 then
            print("⏳ Mất file! Đang chờ tín hiệu... (" .. (30 - timePassed) .. "s)")
            continue -- Vẫn trong thời gian chờ, KHÔNG Hop
        else
            print("⚠️ Hết 30s chờ. Chuyển sang chế độ Server Rác.")
            lastSafeTime = 0 -- Reset về 0 để các lần sau không chờ nữa
        end
    end

    -- [TRƯỜNG HỢP C]: Server Rác
    -- Code sẽ chạy thẳng xuống đây nếu:
    -- 1. Bạn vừa vào game ở server rác (lastSafeTime = 0)
    -- 2. Hoặc đã chờ xong 30s ở trường hợp B
    
    print("🔍 [Server Rác] Đang quét tìm Server ngon để Hop...")
    
    local foundTarget = false
    for _, file in ipairs(files) do
        local success, content = pcall(readfile, file)
        if success then
            local data = HttpService:JSONDecode(content)
            local timeDiff = os.time() - data.timestamp
            
            -- Chỉ hop nếu: File mới + Server vắng + Không phải server hiện tại
            if timeDiff < 180 and data.playerCount <= 11 and data.jobId ~= game.JobId then
                print("🚀 Tìm thấy: " .. data.username .. " -> HOP NGAY!")
                TeleportService:TeleportToPlaceInstance(data.placeId, data.jobId, LocalPlayer)
                foundTarget = true
                break
            end
        end
    end
    
    if not foundTarget then
        print("💤 Chưa tìm thấy server phù hợp...")
    end
end
